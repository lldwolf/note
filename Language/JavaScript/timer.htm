时钟

一个简单的时钟例子

<html>
<head>
<title> test page </title>
<script type="text/javascript">
function display_timer()
{
	var t = document.all["timer"];
	var d = new Date();
	t.innerHTML = d.getYear() + "-" + (d.getMonth() + 1) + "-" + d.getDay() + " " + d.getHours() + ":" + d.getMinutes() + ":" + d.getSeconds();
	setTimeout("display_timer()", 1000);
}

</script>
</head>

<body onload="display_timer()">
<form>
	<p id="timer">当前时间:</p>
</form>
</body>
</html>
