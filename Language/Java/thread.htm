多线程

First View

下例展示了一个最简单的线程使用示例，每隔一秒输出当前时间

package lld.test;

import java.util.Date;

public class ThreadTest {

	public static void main(String[] args) {
		Thread thread = new Thread(new Runnable() {
			
			@Override
			public void run() {
				while (true) {
					try {
						Thread.sleep(1000);
						System.out.println(new Date().toString());
					} catch (Exception e) {
						e.printStackTrace();
					}
				}
			}
		});

		thread.start();
	}

}
