dom4j

用于操作XML文件.

需要引用的包：

  • dom4j-1.6.1.jar
  • jaxen-1.1-beta-6.jar

如果不引用jaxen-1.1-beta-6.jar的话，在调用Document.selectNodes()或Document.selectSingleNode()方法时会出现异常。

一个简单的例子：

import org.dom4j.io.SAXReader;
import org.dom4j.*;

public class TestDom4j
{
	public static void main(String[] args) throws Exception
	{
		try
		{
			SAXReader reader = new SAXReader();
			String xml_file = "D:\\Work\\Java\\INtools\\WebContent\\config\\manage_sql.xml";
			File file = new File(xml_file);
	        Document document = reader.read(xml_file);
	        Node sql_node = document.selectSingleNode("/root/sql[@key=\"list_user\"]");
	        System.out.println(sql_node.getText());
		}catch(Exception ex)
		{
			System.out.print(ex.getMessage());
		}
	}

}
