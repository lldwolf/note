动画

下例展示了一个旋转的太极图，太极图在鼠标移入时旋转，移出后停止旋转

XAML

<UserControl x:Class="WpfDemo.Animation.TaiJiControl"
			 xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
			 xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
			 xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
			 xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
			 mc:Ignorable="d"
			 d:DesignHeight="300"
			 d:DesignWidth="300">
	<UserControl.Resources>
	</UserControl.Resources>
	<Viewbox>
		<Canvas x:Name="taijiCanvas"
				Width="100"
				Height="100"
				RenderTransformOrigin="0.5, 0.5">
			<Canvas.RenderTransform>
				<RotateTransform x:Name="transformer"
								 Angle="0" />
			</Canvas.RenderTransform>
			
			<Canvas.Triggers>
				<EventTrigger RoutedEvent="Canvas.MouseEnter">
					<BeginStoryboard x:Name="rotateAction">
						<Storyboard>
							<DoubleAnimation Storyboard.TargetName="transformer"
											 Storyboard.TargetProperty="Angle"
											 From="0"
											 To="360"
											 SpeedRatio="0.5"
											 AutoReverse="False"
											 RepeatBehavior="Forever"></DoubleAnimation>
						</Storyboard>
					</BeginStoryboard>
				</EventTrigger>
				<EventTrigger RoutedEvent="Canvas.MouseLeave">
					<StopStoryboard BeginStoryboardName="rotateAction" />
				</EventTrigger>
			</Canvas.Triggers>
			
			<Path Fill="Black"
				  Stroke="Black"
				  StrokeThickness="1">
				<Path.Data>
					<GeometryGroup>
						<PathGeometry>
							<PathFigure StartPoint="0, 50"
										IsClosed="False">
								<ArcSegment Point="100, 50"
											Size="25, 25"
											SweepDirection="Clockwise" />
							</PathFigure>
							<PathFigure StartPoint="0, 50"
										IsClosed="False">
								<ArcSegment Point="50, 50"
											Size="25, 25"
											SweepDirection="Clockwise" />
							</PathFigure>
							<PathFigure StartPoint="50, 50"
										IsClosed="False">
								<ArcSegment Point="100, 50"
											Size="25, 25"
											SweepDirection="Counterclockwise" />
							</PathFigure>
						</PathGeometry>
					</GeometryGroup>
				</Path.Data>
			</Path>
			<Path Fill="White"
				  Stroke="Black"
				  StrokeThickness="1">
				<Path.Data>
					<GeometryGroup>
						<PathGeometry>
							<PathFigure StartPoint="0, 50"
										IsClosed="False">
								<ArcSegment Point="100, 50"
											Size="25, 25"
											SweepDirection="Counterclockwise" />
							</PathFigure>
							<PathFigure StartPoint="100, 50"
										IsClosed="False">
								<ArcSegment Point="50, 50"
											Size="25, 25"
											SweepDirection="Clockwise" />
							</PathFigure>
							<PathFigure StartPoint="50, 50"
										IsClosed="False">
								<ArcSegment Point="0, 50"
											Size="25, 25"
											SweepDirection="Counterclockwise" />
							</PathFigure>
						</PathGeometry>
					</GeometryGroup>
				</Path.Data>
			</Path>
			<Path Fill="Black">
				<Path.Data>
					<EllipseGeometry Center="25, 50"
									 RadiusX="5"
									 RadiusY="5" />
				</Path.Data>
			</Path>
			<Path Fill="White">
				<Path.Data>
					<EllipseGeometry Center="75, 50"
									 RadiusX="5"
									 RadiusY="5" />
				</Path.Data>
			</Path>
		</Canvas>
	</Viewbox>
</UserControl>
