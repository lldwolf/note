绘图

下例展示了如何绘制一个太级图，效果如下所示：

XAML

<UserControl x:Class="WpfDemo.Paint.TaiJiControl"
			 xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
			 xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
			 xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
			 xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
			 mc:Ignorable="d"
			 d:DesignHeight="300"
			 d:DesignWidth="300">
	<Viewbox>
		<Canvas x:Name="taijiCanvas"
				Width="100"
				Height="100"
				RenderTransformOrigin="0.5, 0.5">
			<Canvas.RenderTransform>
				<RotateTransform x:Name="transformer"
								 Angle="45" />
			</Canvas.RenderTransform>
			<Path Fill="Black"
				  Stroke="Black"
				  StrokeThickness="1">
				<Path.Data>
					<GeometryGroup>
						<PathGeometry>
							<PathFigure StartPoint="0, 50"
										IsClosed="False">
								<ArcSegment Point="100, 50"
											Size="25, 25"
											SweepDirection="Clockwise" />
							</PathFigure>
							<PathFigure StartPoint="0, 50"
										IsClosed="False">
								<ArcSegment Point="50, 50"
											Size="25, 25"
											SweepDirection="Clockwise" />
							</PathFigure>
							<PathFigure StartPoint="50, 50"
										IsClosed="False">
								<ArcSegment Point="100, 50"
											Size="25, 25"
											SweepDirection="Counterclockwise" />
							</PathFigure>
						</PathGeometry>
					</GeometryGroup>
				</Path.Data>
			</Path>
			<Path Fill="White"
				  Stroke="Black"
				  StrokeThickness="1">
				<Path.Data>
					<GeometryGroup>
						<PathGeometry>
							<PathFigure StartPoint="0, 50"
										IsClosed="False">
								<ArcSegment Point="100, 50"
											Size="25, 25"
											SweepDirection="Counterclockwise" />
							</PathFigure>
							<PathFigure StartPoint="100, 50"
										IsClosed="False">
								<ArcSegment Point="50, 50"
											Size="25, 25"
											SweepDirection="Clockwise" />
							</PathFigure>
							<PathFigure StartPoint="50, 50"
										IsClosed="False">
								<ArcSegment Point="0, 50"
											Size="25, 25"
											SweepDirection="Counterclockwise" />
							</PathFigure>
						</PathGeometry>
					</GeometryGroup>
				</Path.Data>
			</Path>
			<Path Fill="Black">
				<Path.Data>
					<EllipseGeometry Center="25, 50"
									 RadiusX="5"
									 RadiusY="5" />
				</Path.Data>
			</Path>
			<Path Fill="White">
				<Path.Data>
					<EllipseGeometry Center="75, 50"
									 RadiusX="5"
									 RadiusY="5" />
				</Path.Data>
			</Path>
		</Canvas>
	</Viewbox>
</UserControl>
