FTP

常用操作

ftp  # 进入 FTP 环境
open [server_url]				# 打开server

# 输入用户名/密码

binary							# 切换为 binary 模式
put c:/temp/1.zip /tmp/1.zip	# 上传二进制文件
ascii							# 切换为 ascii 模式
put c:/temp/1.txt /tmp/1.txt	# 上传文本文件
get /tmp/1.txt c:/temp/2.txt	# 下载
del /tmp/1.txt					# 删除远端文件
close							# 关闭连接
quit							# 退出 FTP 环境
