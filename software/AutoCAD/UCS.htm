第N章 坐标系统

N.1 将坐标系切换为世界UCS

Private Sub ToWorldUCS()

    '切换UCS至世界UCS
    Dim worldUCS As AcadUCS
    Dim origin(0 To 2) As Double
    Dim xAxisPnt(0 To 2) As Double
    Dim yAxisPnt(0 To 2) As Double
    
    origin(0) = 0#: origin(1) = 0#: origin(2) = 0#
    xAxisPnt(0) = 1#: xAxisPnt(1) = 0#: xAxisPnt(2) = 0#
    yAxisPnt(0) = 0#: yAxisPnt(1) = 1#: yAxisPnt(2) = 0#
    
    Set worldUCS = ThisDrawing.UserCoordinateSystems.Add(origin, xAxisPnt, yAxisPnt, "World_UCS")
    ThisDrawing.ActiveUCS = worldUCS

End Sub
