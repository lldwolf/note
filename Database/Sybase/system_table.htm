Table 信息

-- 根据表名查找表
select * from sysobjects where name like '%table_name%' and type = 'U'

-- 根据列名查找表
select a.name table_name
from sysobjects a, syscolumns b
where a.id = b.id and a.type = 'U' and b.name like 'column_name'

-- 查看表信息
sp_help [table_name]
