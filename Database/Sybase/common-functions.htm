常用函数

日期类

--当前日期
select getdate()

-- 格式分日期 mm/dd/yyyy
select CONVERT(varchar, getdate(), 101)

--取日期的各个部分值
select year(getdate())
select month(getdate())
select day(getdate())
select datepart(year, getdate())
select datepart(month, getdate())
select datepart(day, getdate())
select datepart(hour, getdate())

信息显示类

-- 打印字符串
print 'Hello, world.'

print 后只可能用字符串常量和变量，不支持其它类型和返回值为字符串的函数，如下语句是非法的

-- 不能使用非字符串类型
declare @a datetime
set @a = getdate()
print @a

-- 不能使用的函数返回值, 即使返回值类型为字符串
print convert(varchar, getdate(), 101)
